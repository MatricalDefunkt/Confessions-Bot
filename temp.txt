const runModalCommand = async function (interaction, client) {

	const command = client.commands.get('modal');
	try {
		command.execute(interaction, client);
		return;
	} catch (error) {
		const errGuild = client.guilds.cache.get(`${errGuildId}`);
		const errChannel = errGuild.channels.cache.get(`${errChannelId}`);
		console.error(error); errChannel.send({ content: `An error was caught: \n\`\`\`js\n${error.stack}\`\`\`` });
		return;
	}

}
